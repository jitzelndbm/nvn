    class Client {
        + Note current
        + Note root
        + History history
        + Config config

        + new(Config config) Client

        + get_current() Note
        + set_current(Note new, bool use_history) void
        
        + get_all_notes() List~Note~

        + open_graph() void
        + open_link(Link link) void
    }

    Client --> Note

    class Note {
        + Path path

        + new(Path path) Note
        + remove() void
        + from_template(Template template) Note
        + get_links() List~Link~
        + get_link_cursor(bool backwards) Link
        + evaluate() void
    }

    class Link {
        + int row
        + int column
        + Note note

        + new(TSNode node) Link
    }

    class Template {
        + Path path

        + render() string
    }

    Template --> Note

        + pop() Note
        + push() Note
    }

    History --> Client

    Config --> Client
